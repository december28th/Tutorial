function gaussianelimination
clc
clear
close all
a=[10 3 1 1;1 9 2 1;2 4 8 2;3 2 1 7];
b=[7 -7 4 -5]';
x=zeros(4,1);
x;
for m=1:3
    b(m+1,1)=b(m+1,1)-b(m,1)*a(m+1,m)/a(m,m)
    for i=m+1:4
       for j=m:4
        a(i,j)=a(i,j)-a(m,j)*a(i,m)/a(m,m);
       end
    end
end
a
b
x(4,1)=b(4,1)/a(4,4);
for i=3:-1:1
    x(i,1)=(b(i,1)-sum(a(i,i+1:4)*x(i+1:4,1)))/a(i,i);
end
x
a*x-b
